HTML 30-day challenge - Día 28

Crea una ventana modal que bloquee la interación con otros botones. Requiere un poco de Javascript.

HTML

  
    <dialog id="alerta">
      <h2>¡Censeguido!</h2>
      <p>Hemos superado el desafío #28</p>
      <button onClick="this.parentElement.close()">Cerrar</button>
    </dialog>
    <button id="mostrar">Muéstrame el diálogo</button>
    <script src="./script.js"></script>

JS

  
    const showButton = document.querySelector("#mostrar");
    showButton.addEventListener("click", function () {
      const alertDialog = document.querySelector("#alerta");
      alertDialog.showModal();
    });
